Python 学习 文件处理
	对文件的操作:
		打开文件: open()
			以文本只读模式打开文件:
				file = open('a.txt', 'rt', encoding='utf-8') #字节模式打开时不设置encoding
			以二进制只读模式打开文件:
				file = open('a.txt','rb')
			其余模式请看函数详解open()
		读取文件: 注意:读取操作要使用.close()函数关闭文件对之后才有效
			按字符读取:
				file = open('aa.txt', 'rt') #注意文件读取模式,选用文本只读(or 二进制只读)
				data = file.read(5) #读取5个字符,默认不写参数则读取整个文件;返回字符串
				file.close() #没有这一句data是没有值的
			按行读取:
				file = open("aa.txt", 'rt', encoding='utf-8') 
				while 1:
				  line = file.readline() #使用.readline()函数按行读取
				  if not line:
					break
				  print(line)
				file.close()

	将整个文件目录结构构建成一个字典: 注意: 不要包含中文
		def folder_to_dict(folder_path): 
			#folder_path: 要遍历的文件夹的路径
			#返回一个字典
			dict_path = dict()
			for path, list_folder, list_file in os.walk(folder_path):
				dict_path['path'] = path
				for folder_name in list_folder:
					dict_path[folder_name] = folder_to_dict(path+'\\'+folder_name)
				dict_path['list_file'] = list()
				for list_file_name in list_file:            
					dict_path['list_file'].append(path+'\\'+list_file_name)
				break        
			return dict_path
