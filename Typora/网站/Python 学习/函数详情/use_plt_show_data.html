def use_plt_show_data():	
		#这是个迭代器，使用时使用.send()函数传入数据
		plt.ion()#开启交互模式
		fig = plt.figure(figsize=(16,8), dpi=100)#获取画布
		ax1 = fig.add_subplot(1,2,1)#创建子图1
		ax2 = fig.add_subplot(1,2,2)#创建子图2
		while 1:
			data = yield
			if(data=='over'):
				break
			ax1.cla() #清除子图的图像
			ax1.grid(color='r', linestyle='--', linewidth=1,alpha=0.3)#添加网格线
			ax1.set_title('Title')#标题
			ax1.Xlabel('XAxis')#X轴名称
			ax1.Ylabel('YAxis')#Y轴名称
			ax1.plot(data[0],data[1], label = 'loss')#绘出折线图
			ax1.legend()#显示标签
			ax2.cla() #清除子图的图像
			ax2.grid(color='r', linestyle='--', linewidth=1,alpha=0.3)#添加网格线
			ax2.plot(data[0],data[2], label = 'avg')
			ax2.legend()
			plt.draw() #交互模式下重新绘制图像
			plt.pause(0.1) #保持显示
		plt.ioff()#关闭交互模式
		plt.show()#显示图像
		
	使用示例：
		g = use_plt_show_data()#创建迭代器
		next(g)
		X = list()
		Y1 = list()
		Y2 = list()
		for i range(100):
			X.append(i)
			Y1.append(i*i+3*i+1)
			Y2.append(i*i*i)
			g.send([X,Y1,Y2])
		g.send('over')#结束
